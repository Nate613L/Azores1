const pBor={ok:'rgba(16,185,129,.25)',caution:'rgba(245,158,11,.25)'};
const pCol={ok:'#10b981',caution:'#f59e0b'};

function buildStops(){
  Object.entries(DAYS).forEach(([dayId,day])=>{
    const container=document.getElementById('stops-'+dayId);
    if(!container)return;
    day.stops.forEach((stop,idx)=>{
      const card=document.createElement('button');
      card.className='card';
      card.onclick=()=>openModal(stop,day.accent);
      card.innerHTML=`
        <div class="cnum" style="background:${day.accent}25;border:1.5px solid ${day.accent}77;color:${day.accent}">${idx+1}</div>
        <div class="cthumb">
          <img src="${stop.img}" alt="${stop.name}" onerror="this.style.display='none';this.parentNode.textContent='${stop.emoji}'"/>
        </div>
        <div class="cinfo">
          <div class="cbadges">
            <span class="ctag" style="background:${stop.tagColor}28;color:${stop.tagColor};border:1px solid ${stop.tagColor}44">${stop.tag}</span>
            <span class="ctime">${stop.time}</span>
            <span class="pbadge ${pClass[stop.preg]}">${pLabel[stop.preg]}</span>
          </div>
          <div class="cname">${stop.name}</div>
          <div class="ctype">${stop.emoji} ${stop.type}</div>
        </div>
        <div class="carrow">›</div>`;
      container.appendChild(card);
    });
  });
}

function openModal(stop,accent){
  document.getElementById('memoji').textContent=stop.emoji;
  const img=document.getElementById('mimg');
  img.src=stop.img; img.style.display='block';
  const tag=document.getElementById('mtag');
  tag.textContent=stop.tag;
  tag.style.cssText=`background:${stop.tagColor}33;color:${stop.tagColor};border:1px solid ${stop.tagColor}55;font-size:10px;font-weight:500;padding:3px 9px;border-radius:5px;letter-spacing:.5px`;
  document.getElementById('mtype').textContent=stop.emoji+' '+stop.type;
  document.getElementById('mtitle').textContent=stop.name;
  document.getElementById('mdur').textContent='🕐 '+stop.duration;
  document.getElementById('mdesc').textContent=stop.desc;
  const pe=document.getElementById('mpreg');
  pe.style.cssText=`background:${pBg[stop.preg]};border:1px solid ${pBor[stop.preg]};border-radius:10px;padding:12px 16px;font-size:13px;line-height:1.65;font-weight:300;margin-bottom:16px`;
  pe.innerHTML=`<div style="font-size:10px;letter-spacing:2px;text-transform:uppercase;color:${pCol[stop.preg]};margin-bottom:6px;font-weight:500">🤰 Pregnancy (1–3 months) &nbsp;<span class="pbadge ${pClass[stop.preg]}">${pLabel[stop.preg]}</span></div>${stop.pregNote}`;
  document.getElementById('mtips').textContent=stop.tips;
  document.getElementById('overlay').classList.add('open');
  document.body.style.overflow='hidden';
}

function closeModal(){
  document.getElementById('overlay').classList.remove('open');
  document.body.style.overflow='';
}

function switchDay(dayId){
  document.querySelectorAll('.panel').forEach(p=>p.classList.remove('active'));
  document.getElementById('panel-'+dayId).classList.add('active');
  const order=['mon','tue','wed','extra'];
  const classMap={mon:'a-mon',tue:'a-tue',wed:'a-wed',extra:'a-extra'};
  document.querySelectorAll('.tab').forEach((t,i)=>{
    t.className='tab';
    if(order[i]===dayId) t.classList.add(classMap[dayId]);
  });
}

document.addEventListener('keydown',e=>{if(e.key==='Escape')closeModal();});
